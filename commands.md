# COMMANDS

## Change the map
rcon changelevel de_mirage

## Restart the round
rcon mp_restartgame 1

## Add bots
rcon bot_add_ct
rcon bot_add_t

## Kick all bots
rcon bot_kick

## Change to competitive mode
rcon game_mode 1
rcon game_type 0

## Pause/unpause a match
rcon mp_pause_match
rcon mp_unpause_match

## View server status
rcon status
rcon stats
